# 错因分析

```dataviewjs
const pages = dv.pages('"01_错题"').where(p => p.type === "wrong_problem");

function listify(value) {
  if (!value) return ["待补充"];
  if (Array.isArray(value)) return value.map(String);
  if (typeof value.array === "function") return value.array().map(String);
  return [String(value)];
}

const reasonCounts = new Map();
const methodCounts = new Map();
const knowledgeCounts = new Map();

for (const p of pages) {
  for (const item of listify(p.wrong_reason)) reasonCounts.set(item, (reasonCounts.get(item) ?? 0) + 1);
  for (const item of listify(p.methods)) methodCounts.set(item, (methodCounts.get(item) ?? 0) + 1);
  for (const item of listify(p.knowledge_points)) knowledgeCounts.set(item, (knowledgeCounts.get(item) ?? 0) + 1);
}

function table(title, counts) {
  dv.header(2, title);
  const rows = [...counts.entries()].sort((a, b) => b[1] - a[1]).map(([k, v]) => [k, v]);
  dv.table(["项目", "次数"], rows);
}

table("高频错因", reasonCounts);
table("高频方法", methodCounts);
table("高频知识点", knowledgeCounts);
```
