# 知识点地图

```dataviewjs
const pages = dv.pages('"01_错题"').where(p => p.type === "wrong_problem");

function listify(value) {
  if (!value) return ["待补充"];
  if (Array.isArray(value)) return value.map(String);
  if (typeof value.array === "function") return value.array().map(String);
  return [String(value)];
}

function countField(field) {
  const counts = new Map();
  for (const p of pages) {
    for (const item of listify(p[field])) {
      counts.set(item, (counts.get(item) ?? 0) + 1);
    }
  }
  return [...counts.entries()].sort((a, b) => b[1] - a[1]);
}

dv.header(2, "高频知识点");
dv.table(["知识点", "错题数"], countField("knowledge_points"));

dv.header(2, "高频方法");
dv.table(["方法", "错题数"], countField("methods"));
```
