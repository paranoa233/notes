# 待复习

```dataviewjs
const today = dv.date("today");
const pages = dv.pages('"01_错题"')
  .where(p => p.type === "wrong_problem")
  .where(p => {
    const status = String(p.status ?? "");
    const next = p.next_review ? dv.date(p.next_review) : null;
    return status === "待复习" || status === "仍错" || (next && next <= today);
  })
  .sort(p => p.next_review ?? "9999-12-31", "asc")
  .sort(p => Number(p.mastery ?? 9), "asc");

dv.table(
  ["错题", "专题", "小节", "状态", "掌握度", "错因", "下次复习"],
  pages.map(p => [
    p.file.link,
    p.module ?? "待补充",
    p.topic ?? "待补充",
    p.status ?? "待复习",
    p.mastery ?? "",
    p.wrong_reason ?? "",
    p.next_review ?? ""
  ])
);
```
